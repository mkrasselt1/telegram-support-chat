Integration snippet

<!-- Place before </body> -->
<script>
window.SupportChatConfig = {
  // Required: URL to your chat.php endpoint
  endpoint: '/support-chat/chat.php',

  // Optional: path to service worker (must be at or above widget scope)
  swPath: '/sw.js',

  // Optional: path to CSS (auto-detected from script src if omitted)
  cssPath: '/support-chat/support-chat.css',

  // Optional: logged-in user info (sent to Telegram as context)
  user: {
    name:  'Jane Doe',
    email: 'jane@example.com',
    id:    '42',
  },

  // Optional: polling interval in ms (default: 4000)
  pollInterval: 4000,

  // Optional: theme overrides
  theme: {
    primary: '#0088cc',   // Your brand color
    radius:  '16px',
  },
};
</script>
<script src="/support-chat/support-chat.js" async></script>
